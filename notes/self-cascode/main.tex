\begin{equation}
V_{\mathrm{DS}} = V_{\mathrm{D}} - V_{\mathrm{S}} = V_{\mathrm{DS,1}} + V_{\mathrm{DS,2}} 
\end{equation}
\begin{equation}
V_{\mathrm{GS,1}} = V_{\mathrm{GS}}
\end{equation}
\begin{equation}
V_{\mathrm{GS,2}} = V_{\mathrm{GS}} - V_{\mathrm{DS,1}}
\end{equation}
\begin{equation}
V_{\mathrm{DS,1}} = V_{\mathrm{X}} - V_{\mathrm{S}}
\end{equation}
\begin{equation}
V_{\mathrm{DS,2}} = V_{\mathrm{D}} - V_{\mathrm{X}}
\end{equation}
\begin{equation}
I_{\mathrm{D}} = I_{\mathrm{D,1}} = I_{\mathrm{D,2}}
\end{equation}

We utilize the square-law \cite{mosfet-square-law}
\begin{equation}
\beta_{1} = K' \cdot S_{1} = K'  \frac{W_{1}}{L_{1}}
\end{equation}
\begin{equation}
\beta_{2} = K' \cdot S_{2} = K'  \frac{W_{2}}{L_{2}}
\end{equation}
without any body biasing
\begin{equation}
V_{\mathrm{TH}} = V_{\mathrm{TH,1}} = V_{\mathrm{TH,2}}.
\end{equation}
