V_{\mathrm{GS}} = V_{\mathrm{S}} - V_{\mathrm{S}}
\end{equation}

\begin{equation}\label{eq:vds}
V_{\mathrm{DS}} = V_{\mathrm{D}} - V_{\mathrm{S}}
\end{equation}

\begin{equation}\label{eq:vov}
V_{\mathrm{OV}} = V_{\mathrm{GS}} - V_{\mathrm{T}}
\end{equation}

\begin{equation}\label{eq:vm}
V_{\mathrm{M}} = V_{\mathrm{DS}} - V_{\mathrm{OV}}
\end{equation}

The reported capacitances are rather odd, because they are sometimes
negative.
This is why we will denote them with a calligraphic $\mathcal{C}$.

\medskip

When we arrange the capacitances as a matrix
\begin{equation}
\begin{bmatrix}
\mathcal{C}_{\mathrm{dd}} & \mathcal{C}_{\mathrm{gd}} & \mathcal{C}_{\mathrm{sd}} & \mathcal{C}_{\mathrm{bd}} \\
\mathcal{C}_{\mathrm{dg}} & \mathcal{C}_{\mathrm{gg}} & \mathcal{C}_{\mathrm{sg}} & \mathcal{C}_{\mathrm{bg}} \\
\mathcal{C}_{\mathrm{ds}} & \mathcal{C}_{\mathrm{gs}} & \mathcal{C}_{\mathrm{ss}} & \mathcal{C}_{\mathrm{bs}} \\
\mathcal{C}_{\mathrm{db}} & \mathcal{C}_{\mathrm{gb}} & \mathcal{C}_{\mathrm{sb}} & \mathcal{C}_{\mathrm{bb}} \\
\end{bmatrix}
\end{equation}
we can identify the following characteristics:
\begin{itemize}
  \item $\mathcal{C}_{ii}$ is always positive.
  \item Capacitances are not reciprocal, i.e. $\mathcal{C}_{ij}\neq\mathcal{C}_{ji}$
    nor $\mathcal{C}_{ij}\neq-\mathcal{C}_{ji}$.
  \item The capacitances in each column or row add up approximately to zero 
    (Tab.~\ref{tab:summ-cap}, Fig.~\ref{fig:cap-sum-vis}).
\end{itemize}

\begin{equation}\label{eq:charge}
Q_{\mathrm{D}}+Q_{\mathrm{S}}+Q_{\mathrm{G}}+Q_{\mathrm{B}}=0.
\end{equation}

The calligraphic $\mathcal{C}$ is used to denote transcapitances.

\begin{equation}\label{eq:capacitance}
\mathcal{C}_{ij}=\frac{\partial Q_{i}}{\partial V_{j}},
\end{equation}


\begin{equation}\label{eq:dgl-g-1}
i_{\mathrm{G}} = 
  \mathcal{C}_{\mathrm{gg}} \dv{v_{\mathrm{G}}}{t} 
+ \mathcal{C}_{\mathrm{gd}} \dv{v_{\mathrm{D}}}{t} 
+ \mathcal{C}_{\mathrm{gs}} \dv{v_{\mathrm{S}}}{t} 
+ \mathcal{C}_{\mathrm{gb}} \dv{v_{\mathrm{B}}}{t}
\end{equation}
\begin{equation}\label{eq:dgl-d-1}
i_{\mathrm{D}} = 
  \mathcal{C}_{\mathrm{dg}} \dv{v_{\mathrm{G}}}{t} 
+ \mathcal{C}_{\mathrm{dd}} \dv{v_{\mathrm{D}}}{t} 
+ \mathcal{C}_{\mathrm{ds}} \dv{v_{\mathrm{S}}}{t} 
+ \mathcal{C}_{\mathrm{db}} \dv{v_{\mathrm{B}}}{t}
\end{equation}
\begin{equation}\label{eq:dgl-s-1}
i_{\mathrm{S}} = 
  \mathcal{C}_{\mathrm{sg}} \dv{v_{\mathrm{G}}}{t} 
+ \mathcal{C}_{\mathrm{sd}} \dv{v_{\mathrm{D}}}{t} 
+ \mathcal{C}_{\mathrm{ss}} \dv{v_{\mathrm{S}}}{t} 
+ \mathcal{C}_{\mathrm{sb}} \dv{v_{\mathrm{B}}}{t}
\end{equation}
\begin{equation}\label{eq:dgl-b-1}
i_{\mathrm{B}} = 
  \mathcal{C}_{\mathrm{bg}} \dv{v_{\mathrm{G}}}{t} 
+ \mathcal{C}_{\mathrm{bd}} \dv{v_{\mathrm{D}}}{t} 
+ \mathcal{C}_{\mathrm{bs}} \dv{v_{\mathrm{S}}}{t} 
+ \mathcal{C}_{\mathrm{bb}} \dv{v_{\mathrm{B}}}{t}
\end{equation}

With the continuation equation%
\footnote{Obtained by deriving \eqref{eq:charge}.} 
\begin{equation}\label{eq:cont-eq}
i_{\mathrm{G}}+i_{\mathrm{D}}+i_{\mathrm{S}}+i_{\mathrm{B}}=0
\end{equation}
and \cref{eq:dgl-g-1,eq:dgl-d-1,eq:dgl-s-1,eq:dgl-b-1} follows
\begin{equation}\label{eq:cap-sum-1}
\sum_{j} \mathcal{C}_{ij} = 0. 
\end{equation}

When we set $\dv{v_{\mathrm{i}}}{t}\neq0$ and $\dv{v_{\mathrm{j}}}{t}=0$
with $j\neq i$ follows with \eqref{eq:cont-eq}
\begin{equation}\label{eq:cap-sum-2}
\sum_{i} \mathcal{C}_{ij} = 0. 
\end{equation}
